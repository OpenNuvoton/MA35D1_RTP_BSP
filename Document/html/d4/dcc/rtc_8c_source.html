99void RTC_GetDateAndTime(S_RTC_TIME_DATA_T *sPt)
100{
101 uint32_t u32Tmp;
102
103 sPt->u32TimeScale = RTC->CLKFMT & RTC_CLKFMT_24HEN_Msk; /* 12/24-hour */
104 sPt->u32DayOfWeek = RTC->WEEKDAY & RTC_WEEKDAY_WEEKDAY_Msk; /* Day of the week */
105
106 /* Get [Date digit] data */
107 g_u32hiYear = (RTC->CAL & RTC_CAL_TENYEAR_Msk) >> RTC_CAL_TENYEAR_Pos;
108 g_u32loYear = (RTC->CAL & RTC_CAL_YEAR_Msk) >> RTC_CAL_YEAR_Pos;
109 g_u32hiMonth = (RTC->CAL & RTC_CAL_TENMON_Msk) >> RTC_CAL_TENMON_Pos;
110 g_u32loMonth = (RTC->CAL & RTC_CAL_MON_Msk) >> RTC_CAL_MON_Pos;
111 g_u32hiDay = (RTC->CAL & RTC_CAL_TENDAY_Msk) >> RTC_CAL_TENDAY_Pos;
112 g_u32loDay = (RTC->CAL & RTC_CAL_DAY_Msk) >> RTC_CAL_DAY_Pos;
113
114 /* Get [Time digit] data */
115 g_u32hiHour = (RTC->TIME & RTC_TIME_TENHR_Msk) >> RTC_TIME_TENHR_Pos;
116 g_u32loHour = (RTC->TIME & RTC_TIME_HR_Msk) >> RTC_TIME_HR_Pos;
117 g_u32hiMin = (RTC->TIME & RTC_TIME_TENMIN_Msk) >> RTC_TIME_TENMIN_Pos;
118 g_u32loMin = (RTC->TIME & RTC_TIME_MIN_Msk) >> RTC_TIME_MIN_Pos;
119 g_u32hiSec = (RTC->TIME & RTC_TIME_TENSEC_Msk) >> RTC_TIME_TENSEC_Pos;
120 g_u32loSec = (RTC->TIME & RTC_TIME_SEC_Msk) >> RTC_TIME_SEC_Pos;
121
122 /* Compute to 20XX year */
123 u32Tmp = (g_u32hiYear * 10ul);
124 u32Tmp += g_u32loYear;
125 sPt->u32Year = u32Tmp + RTC_YEAR2000;
126
127 /* Compute 0~12 month */
128 u32Tmp = (g_u32hiMonth * 10ul);
129 sPt->u32Month = u32Tmp + g_u32loMonth;
130
131 /* Compute 0~31 day */
132 u32Tmp = (g_u32hiDay * 10ul);
133 sPt->u32Day = u32Tmp + g_u32loDay;
134
135 /* Compute 12/24 hour */
136 if(sPt->u32TimeScale == RTC_CLOCK_12)
137 {
138 u32Tmp = (g_u32hiHour * 10ul);
139 u32Tmp += g_u32loHour;
140 sPt->u32Hour = u32Tmp; /* AM: 1~12. PM: 21~32. */
141
142 if(sPt->u32Hour >= 21ul)
143 {
144 sPt->u32AmPm = RTC_PM;
145 sPt->u32Hour -= 20ul;
146 }
147 else
148 {
149 sPt->u32AmPm = RTC_AM;
150 }
151
152 u32Tmp = (g_u32hiMin * 10ul);
153 u32Tmp += g_u32loMin;
154 sPt->u32Minute = u32Tmp;
155
156 u32Tmp = (g_u32hiSec * 10ul);
157 u32Tmp += g_u32loSec;
158 sPt->u32Second = u32Tmp;
159 }
160 else
161 {
162 u32Tmp = (g_u32hiHour * 10ul);
163 u32Tmp += g_u32loHour;
164 sPt->u32Hour = u32Tmp;
165
166 u32Tmp = (g_u32hiMin * 10ul);

292void RTC_SetDateAndTime(S_RTC_TIME_DATA_T *sPt)
293{
294 uint32_t u32RegCAL, u32RegTIME;
295
296 sPt->u32TimeScale = (RTC->CLKFMT & RTC_CLKFMT_24HEN_Msk);
297
298 /*-----------------------------------------------------------------------------------------------------*/
299 /* Set RTC 24/12 hour setting and Day of the Week */
300 /*-----------------------------------------------------------------------------------------------------*/
301 if(sPt->u32TimeScale == RTC_CLOCK_12)
302 {
303 /*-------------------------------------------------------------------------------------------------*/
304 /* Important, range of 12-hour PM mode is 21 up to 32 */
305 /*-------------------------------------------------------------------------------------------------*/
306 if(sPt->u32AmPm == RTC_PM)
307 {
308 sPt->u32Hour += 20ul;
309 }
310 }
311
312 /* Set Day of the Week */
313 RTC->WEEKDAY = sPt->u32DayOfWeek;
314
315 /*-----------------------------------------------------------------------------------------------------*/
316 /* Set RTC Current Date and Time */
317 /*-----------------------------------------------------------------------------------------------------*/
318 u32RegCAL = ((sPt->u32Year - RTC_YEAR2000) / 10ul) << 20;
319 u32RegCAL |= (((sPt->u32Year - RTC_YEAR2000) % 10ul) << 16);
320 u32RegCAL |= ((sPt->u32Month / 10ul) << 12);
321 u32RegCAL |= ((sPt->u32Month % 10ul) << 8);
322 u32RegCAL |= ((sPt->u32Day / 10ul) << 4);
323 u32RegCAL |= (sPt->u32Day % 10ul);
324
325 u32RegTIME = ((sPt->u32Hour / 10ul) << 20);
326 u32RegTIME |= ((sPt->u32Hour % 10ul) << 16);
327 u32RegTIME |= ((sPt->u32Minute / 10ul) << 12);
328 u32RegTIME |= ((sPt->u32Minute % 10ul) << 8);
329 u32RegTIME |= ((sPt->u32Second / 10ul) << 4);
330 u32RegTIME |= (sPt->u32Second % 10ul);
331
332 /*-----------------------------------------------------------------------------------------------------*/
333 /* Set RTC Calender and Time Loading */
334 /*-----------------------------------------------------------------------------------------------------*/
335 RTC->CAL = (uint32_t)u32RegCAL;
336 RTC->TIME = (uint32_t)u32RegTIME;
337}
