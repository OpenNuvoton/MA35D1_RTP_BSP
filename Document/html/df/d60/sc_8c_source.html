35uint32_t SC_IsCardInserted(SC_T *sc)
36{
37 uint32_t ret;
38 /* put conditions into two variable to remove IAR compilation warning */
39 uint32_t cond1 = ((sc->STATUS & SC_STATUS_CDPINSTS_Msk) >> SC_STATUS_CDPINSTS_Pos);
40 uint32_t cond2 = ((sc->CTL & SC_CTL_CDLV_Msk) >> SC_CTL_CDLV_Pos);
41
42 if((sc == SC0) && (u32CardStateIgnore[0] == 1UL))
43 {
44 ret = (uint32_t)TRUE;
45 }
46 else if((sc == SC1) && (u32CardStateIgnore[1] == 1UL))
47 {
48 ret = (uint32_t)TRUE;
49 }
50 else if(cond1 != cond2)
51 {
52 ret = (uint32_t)FALSE;
53 }
54 else
55 {
56 ret = (uint32_t)TRUE;
57 }
58 return ret;
59}

143void SC_ResetReader(SC_T *sc)
144{
145 uint32_t u32Intf;
146
147 if(sc == SC0)
148 {
149 u32Intf = 0UL;
150 }
151 else
152 {
153 u32Intf = 1UL;
154 }
155
156 /* Reset FIFO, enable auto de-activation while card removal */
157 sc->ALTCTL |= (SC_ALTCTL_TXRST_Msk | SC_ALTCTL_RXRST_Msk | SC_ALTCTL_ADACEN_Msk);
158 /* Set Rx trigger level to 1 character, longest card detect debounce period, disable error retry (EMV ATR does not use error retry) */
159 while(sc->CTL & SC_CTL_SYNC_Msk)
160 {
161 ;
162 }
163 sc->CTL &= ~(SC_CTL_RXTRGLV_Msk |
164 SC_CTL_CDDBSEL_Msk |
165 SC_CTL_TXRTY_Msk |
166 SC_CTL_TXRTYEN_Msk |
167 SC_CTL_RXRTY_Msk |
168 SC_CTL_RXRTYEN_Msk);
169 while(sc->CTL & SC_CTL_SYNC_Msk)
170 {
171 ;
172 }
173 /* Enable auto convention, and all three smartcard internal timers */
174 sc->CTL |= SC_CTL_AUTOCEN_Msk | SC_CTL_TMRSEL_Msk;
175 /* Disable Rx timeout */
176 sc->RXTOUT = 0UL;
177 /* 372 clocks per ETU by default */
178 sc->ETUCTL= 371UL;
179
180
181 /* Enable necessary interrupt for smartcard operation */
182 if(u32CardStateIgnore[u32Intf]) /* Do not enable card detect interrupt if card present state ignore */
183 {
184 sc->INTEN = (SC_INTEN_RDAIEN_Msk |
185 SC_INTEN_TERRIEN_Msk |
186 SC_INTEN_TMR0IEN_Msk |
187 SC_INTEN_TMR1IEN_Msk |
188 SC_INTEN_TMR2IEN_Msk |
189 SC_INTEN_BGTIEN_Msk |
190 SC_INTEN_ACERRIEN_Msk);
191 }
192 else
193 {
194 sc->INTEN = (SC_INTEN_RDAIEN_Msk |
195 SC_INTEN_TERRIEN_Msk |
196 SC_INTEN_TMR0IEN_Msk |
197 SC_INTEN_TMR1IEN_Msk |
198 SC_INTEN_TMR2IEN_Msk |
199 SC_INTEN_BGTIEN_Msk |
200 SC_INTEN_CDIEN_Msk |
201 SC_INTEN_ACERRIEN_Msk);
202 }

333uint32_t SC_GetInterfaceClock(SC_T *sc)
334{
335 uint32_t u32ClkSrc, u32Num, u32Clk;
336
337 if(sc == SC0)
338 {
339 u32Num = 0UL;
340 }
341 else
342 {
343 u32Num = 1UL;
344 }
345
346 u32ClkSrc = CLK->CLKSEL4 >> (u32Num + CLK_CLKSEL4_SC0SEL_Pos);
347
348 /* Get smartcard module clock */
349 if(u32ClkSrc == 0UL)
350 {
351 u32Clk = __HXT;
352 }
353 else
354 {
355 u32Clk = CLK_GetPCLK3Freq();
356 }
357
358 u32Clk /= (((CLK->CLKDIV1 >> (4UL * u32Num)) & CLK_CLKDIV1_SC0DIV_Msk) + 1UL) * 1000UL;;
359
360 return u32Clk;
