41uint32_t TIMER_Open(TIMER_T *timer, uint32_t u32Mode, uint32_t u32Freq)
42{
43 uint32_t u32Clk = TIMER_GetModuleClock(timer);
44 uint32_t u32Cmpr = 0UL, u32Prescale = 0UL;
45
46 /* Fastest possible timer working freq is (u32Clk / 2). While cmpr = 2, prescaler = 0. */
47 if (u32Freq > (u32Clk / 2UL))
48 {
49 u32Cmpr = 2UL;
50 }
51 else
52 {
53 u32Cmpr = u32Clk / u32Freq;
54 u32Prescale = (u32Cmpr >> 24); /* for 24 bits CMPDAT */
55
56 if (u32Prescale > 0UL)
57 u32Cmpr = u32Cmpr / (u32Prescale + 1UL);
58 }
59
60 timer->CTL = u32Mode | u32Prescale;
61 timer->CMP = u32Cmpr;
62
63 return (u32Clk / (u32Cmpr * (u32Prescale + 1UL)));
64}

93void TIMER_Delay(TIMER_T *timer, uint32_t u32Usec)
94{
95 uint32_t u32Clk = TIMER_GetModuleClock(timer);
96 uint32_t u32Prescale = 0UL, delay = (SystemCoreClock / u32Clk) + 1UL;
97 uint32_t u32Cmpr, u32NsecPerTick;
98
99 /* Clear current timer configuration */
100 timer->CTL = 0UL;
101 timer->EXTCTL = 0UL;
102
103 if (u32Clk <= 1000000UL) /* min delay is 1000 us if timer clock source is <= 1 MHz */
104 {
105 if (u32Usec < 1000UL)
106 {
107 u32Usec = 1000UL;
108 }
109
110 if (u32Usec > 1000000UL)
111 {
112 u32Usec = 1000000UL;
113 }
114 }
115 else
116 {
117 if (u32Usec < 100UL)
118 {
119 u32Usec = 100UL;
120 }
121
122 if (u32Usec > 1000000UL)
123 {
124 u32Usec = 1000000UL;
125 }
126 }
127
128 if (u32Clk <= 1000000UL)
129 {
130 u32Prescale = 0UL;
131 u32NsecPerTick = 1000000000UL / u32Clk;
132 u32Cmpr = (u32Usec * 1000UL) / u32NsecPerTick;
133 }
134 else
135 {
136 u32Cmpr = u32Usec * (u32Clk / 1000000UL);
137 u32Prescale = (u32Cmpr >> 24); /* for 24 bits CMPDAT */
138
139 if (u32Prescale > 0UL)
140 u32Cmpr = u32Cmpr / (u32Prescale + 1UL);
141 }
142
143 timer->CMP = u32Cmpr;
144 timer->CTL = TIMER_CTL_CNTEN_Msk | TIMER_ONESHOT_MODE | u32Prescale;
145
146 /* When system clock is faster than timer clock, it is possible timer active bit cannot set in time while we check it.
147 And the while loop below return immediately, so put a tiny delay here allowing timer start counting and raise active flag. */
148 for (; delay > 0UL; delay--)
149 {
150 __NOP();
151 }
152
153 while (timer->CTL & TIMER_CTL_ACTSTS_Msk)
154 {
155 ;
156 }
157}
