55uint32_t TPWM_ConfigOutputFreqAndDuty(TIMER_T *timer, uint32_t u32Frequency, uint32_t u32DutyCycle)
56{
57 uint32_t u32PWMClockFreq, u32TargetFreq;
58 uint32_t u32Prescaler = 0x1000UL, u32Period, u32CMP;
59
60 u32PWMClockFreq = CLK_GetPLLClockFreq(SYSPLL);
61
62 /* Calculate u16PERIOD and u16PSC */
63 for (u32Prescaler = 1UL; u32Prescaler <= 0x1000UL; u32Prescaler++)
64 {
65 u32Period = (u32PWMClockFreq / u32Prescaler) / u32Frequency;
66
67 /* If target u32Period is larger than 0x10000, need to use a larger prescaler */
68 if (u32Period <= 0x10000UL)
69 {
70 break;
71 }
72 }
73
74 /* Store return value here 'cos we're gonna change u32Prescaler & u32Period to the real value to fill into register */
75 u32TargetFreq = (u32PWMClockFreq / u32Prescaler) / u32Period;
76
77 /* Set PWM to up count type */
78 timer->PWMCTL = (timer->PWMCTL & ~TIMER_PWMCTL_CNTTYPE_Msk) | (TPWM_UP_COUNT << TIMER_PWMCTL_CNTTYPE_Pos);
79
80 /* Set PWM to auto-reload mode */
81 timer->PWMCTL = (timer->PWMCTL & ~TIMER_PWMCTL_CNTMODE_Msk) | TPWM_AUTO_RELOAD_MODE;
82
83 /* Convert to real register value */
84 TPWM_SET_PRESCALER(timer, (u32Prescaler - 1UL));
85
86 TPWM_SET_PERIOD(timer, (u32Period - 1UL));
87
88 if (u32DutyCycle)
89 {
90 u32CMP = (u32DutyCycle * u32Period) / 100UL;
91 }
92 else
93 {
94 u32CMP = 0UL;
95 }
96
97 TPWM_SET_CMPDAT(timer, u32CMP);
98
99 return (u32TargetFreq);
100}
