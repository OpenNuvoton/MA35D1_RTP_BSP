1035uint32_t CLK_GetPLLClockFreq(uint32_t u32PllIdx)
1036{
1037 uint32_t u32M, u32N, u32P, u32X, u32Mode;
1038 uint32_t u32CTLVal0, u32CTLVal1, u32PllClk, u32RefClk;
1039
1040 u32Mode = CLK_GetPLLOpMode(u32PllIdx);
1041
1042 u32CTLVal0 = *(volatile uint32_t *)(CLK_BASE + 0x60 + (0x10 * (u32PllIdx)));
1043 u32CTLVal1 = *(volatile uint32_t *)(CLK_BASE + 0x64 + (0x10 * (u32PllIdx)));
1044
1045 if (u32Mode == 0)
1046 {
1047 u32N = (u32CTLVal0 & CLK_PLL0CTL0_FBDIV_Msk);
1048 u32M = (u32CTLVal0 & CLK_PLL0CTL0_INDIV_Msk) >> (CLK_PLL0CTL0_INDIV_Pos);
1049 u32P = (u32CTLVal1 & CLK_PLL0CTL1_OUTDIV_Msk) >> (CLK_PLL0CTL1_OUTDIV_Pos);
1050
1051 /* u32RefClk is shifted to avoid overflow */
1052 u32RefClk = __HXT / 100;
1053 /* Actual PLL output clock frequency */
1054 u32PllClk = ((u32RefClk * u32N) / (u32P * u32M)) * 100;
1055
1056 }
1057 else if (u32Mode == 1)
1058 {
1059 u32N = (u32CTLVal0 & CLK_PLL0CTL0_FBDIV_Msk);
1060 u32M = (u32CTLVal0 & CLK_PLL0CTL0_INDIV_Msk) >> (CLK_PLL0CTL0_INDIV_Pos);
1061 u32P = (u32CTLVal1 & CLK_PLL0CTL1_OUTDIV_Msk) >> (CLK_PLL0CTL1_OUTDIV_Pos);
1062 u32X = (u32CTLVal1 & CLK_PLL0CTL1_FRAC_Msk) >> (CLK_PLL0CTL1_FRAC_Pos);
1063
1064 /* Actual PLL output clock frequency */
1065 u32X = (((u32X * 1000) + 500) >> 24);
1066 u32PllClk = (__HXT * ((u32N * 1000) + u32X)) / 1000 / u32P / u32M;
1067 }
1068 else
1069 {
1070 u32N = (u32CTLVal0 & CLK_PLL0CTL0_FBDIV_Msk);
1071 u32M = (u32CTLVal0 & CLK_PLL0CTL0_INDIV_Msk) >> (CLK_PLL0CTL0_INDIV_Pos);
1072 //u32SR = (u32CTLVal0 & CLK_PLL0CTL0_SSRATE_Msk) >> (CLK_PLL0CTL0_SSRATE_Pos);
1073
1074 u32P = (u32CTLVal1 & CLK_PLL0CTL1_OUTDIV_Msk) >> (CLK_PLL0CTL1_OUTDIV_Pos);
1075 u32X = (u32CTLVal1 & CLK_PLL0CTL1_FRAC_Msk) >> (CLK_PLL0CTL1_FRAC_Pos);
1076
1077 /* Actual PLL output clock frequency */
1078 u32X = ((u32X * 1000) >> 24);
1079 u32PllClk = (__HXT * ((u32N * 1000) + u32X)) / 1000 / u32P / u32M;
1080 }
1081
1082 return u32PllClk;
1083}
