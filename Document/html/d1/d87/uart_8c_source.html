181void UART_Open(UART_T* uart, uint32_t u32baudrate)
182{
183 uint32_t u32UartClkSrcSel=0ul, u32UartClkDivNum=0ul;
184 uint32_t u32ClkTbl[4] = {__HXT, 0ul, __LXT, __HIRC};
185 uint32_t u32Baud_Div = 0ul;
186 uint32_t u32UartPort;
187
188 if(uart==(UART_T*)UART16)
189 {
190 u32UartPort = 16;
191 }
192 else
193 {
194 u32UartPort = ((uint32_t)uart & 0xf0000) >> 16;
195 }
196
197 if((u32UartPort >= 1) && (u32UartPort <= 7))
198 {
199 u32UartClkSrcSel = ((CLK->CLKSEL2 >> (16 + u32UartPort*2)) & 0x3ul);
200
201 if((u32UartPort >= 1) && (u32UartPort <= 3))
202 {
203 u32UartClkDivNum = ((CLK->CLKDIV1 >> ((u32UartPort+4)*4)) & 0xful);
204 }
205 else
206 {
207 u32UartClkDivNum = ((CLK->CLKDIV2 >> ((u32UartPort-4)*4)) & 0xful);
208 }
209 }
210 else if((u32UartPort >= 8) && (u32UartPort <= 16))
211 {
212 u32UartClkSrcSel = ((CLK->CLKSEL3 >> ((u32UartPort-8)*2)) & 0x3ul);
213
214 if((u32UartPort >= 8) && (u32UartPort <= 11))
215 {
216 u32UartClkDivNum = ((CLK->CLKDIV2 >> ((u32UartPort-4)*4)) & 0xful);
217 }
218 else
219 {
220 u32UartClkDivNum = ((CLK->CLKDIV3 >> ((u32UartPort-12)*4)) & 0xful);
221 }
222 }
223
224 /* Select UART function */
225 uart->FUNCSEL = UART_FUNCSEL_UART;
226
227 /* Set UART line configuration */
228 uart->LINE = UART_WORD_LEN_8 | UART_PARITY_NONE | UART_STOP_BIT_1;
229
230 /* Set UART Rx and RTS trigger level */
231 uart->FIFO &= ~(UART_FIFO_RFITL_Msk | UART_FIFO_RTSTRGLV_Msk);
232
233 /* Get PLL clock frequency if UART clock source selection is PLL */
234 if(u32UartClkSrcSel == 1ul)
235 {
236 u32ClkTbl[u32UartClkSrcSel] = CLK_GetSYSCLK1Freq();
237 }
238
239 /* Set UART baud rate */
240 if(u32baudrate != 0ul)
241 {
242 u32Baud_Div = UART_BAUD_MODE2_DIVIDER((u32ClkTbl[u32UartClkSrcSel]) / (u32UartClkDivNum + 1ul), u32baudrate);
243
244 if(u32Baud_Div > 0xFFFFul)
245 {
246 uart->BAUD = (UART_BAUD_MODE0 | UART_BAUD_MODE0_DIVIDER((u32ClkTbl[u32UartClkSrcSel]) / (u32UartClkDivNum + 1ul), u32baudrate));
247 }
248 else
249 {
250 uart->BAUD = (UART_BAUD_MODE2 | u32Baud_Div);
251 }
252 }
253}
