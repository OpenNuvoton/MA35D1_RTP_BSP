1151uint32_t SPII2S_Open(SPI_T *i2s, uint32_t u32MasterSlave, uint32_t u32SampleRate, uint32_t u32WordWidth, uint32_t u32Channels, uint32_t u32DataFormat)
1152{
1153 uint32_t u32Divider;
1154 uint32_t u32BitRate, u32SrcClk, u32RetValue;
1155
1156 /* Reset SPI/I2S */
1157 if(i2s == SPI0)
1158 {
1159 SYS->IPRST1 |= SYS_IPRST1_SPI0RST_Msk;
1160 SYS->IPRST1 &= ~SYS_IPRST1_SPI0RST_Msk;
1161 }
1162 else if(i2s == SPI1)
1163 {
1164 SYS->IPRST1 |= SYS_IPRST1_SPI1RST_Msk;
1165 SYS->IPRST1 &= ~SYS_IPRST1_SPI1RST_Msk;

1170 SYS->IPRST1 &= ~SYS_IPRST1_SPI2RST_Msk;
1171 }
1172 else
1173 {

1175 SYS->IPRST2 &= ~SYS_IPRST2_SPI3RST_Msk;
1176 }
1177

1180 /* Set TX FIFO threshold to 2 and RX FIFO threshold to 1 */
1181 SPI_SetFIFO(i2s, 2, 1);
1182
1183 if(u32MasterSlave == SPI_MASTER)
1184 {
1185 /* Get the source clock rate */
1186 u32SrcClk = SPII2S_GetSourceClockFreq(i2s);

1199

1205 }
1206 else

1209 i2s->I2SCLK &= ~SPI_I2SCLK_BCLKDIV_Msk;
1210
1211 if(i2s == SPI0)
1212 {

1214 CLK->CLKSEL4 = (CLK->CLKSEL4 & (~CLK_CLKSEL4_SPI0SEL_Msk)) | CLK_CLKSEL4_SPI0SEL_PCLK1;
1215 /* Enable TX function, RX function and I2S mode. */
1216 i2s->I2SCTL |= (SPI_I2SCTL_RXEN_Msk | SPI_I2SCTL_TXEN_Msk | SPI_I2SCTL_I2SEN_Msk);

1219 }

1221 {
