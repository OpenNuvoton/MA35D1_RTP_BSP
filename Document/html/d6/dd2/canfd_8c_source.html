407static void CANFD_GetSegments(uint32_t u32NominalBaudRate, uint32_t u32DataBaudRate, uint32_t u32Ntq, uint32_t u32Dtq, CANFD_TIMEING_CONFIG_T *psConfig)
408{
409 float ideal_sp;
410 int int32P1;
411
412 /* get ideal sample point */
413 if (u32NominalBaudRate >= 1000000) ideal_sp = 0.750;
414 else if (u32NominalBaudRate >= 800000) ideal_sp = 0.800;
415 else ideal_sp = 0.875;
416
417 /* distribute time quanta */
418 int32P1 = (int)(u32Ntq * ideal_sp);
419 /* can controller doesn't separate prop seg and phase seg 1 */
420 psConfig->u8NominalPropSeg = 0;
421 /* subtract one TQ for sync seg */
422 psConfig->u8NominalPhaseSeg1 = int32P1 - 1;
423 psConfig->u8NominalPhaseSeg2 = u32Ntq - int32P1;
424 /* sjw is 20% of total TQ, rounded to nearest int */
425 psConfig->u8NominalRJumpwidth = (u32Ntq + (5 - 1)) / 5;
426
427 /* if using baud rate switching then distribute time quanta for data rate */
428 if (u32Dtq > 0)
429 {
430 /* get ideal sample point */
431 if (u32DataBaudRate >= 1000000) ideal_sp = 0.750;
432 else if (u32DataBaudRate >= 800000) ideal_sp = 0.800;
433 else ideal_sp = 0.875;
434
435 /* distribute time quanta */
436 int32P1 = (int)(u32Dtq * ideal_sp);
437 /* can controller doesn't separate prop seg and phase seg 1 */
438 psConfig->u8DataPropSeg = 0;
439 /* subtract one TQ for sync seg */
440 psConfig->u8DataPhaseSeg1 = int32P1 - 1;
441 psConfig->u8DataPhaseSeg2 = u32Dtq - int32P1;
442 /* sjw is 20% of total TQ, rounded to nearest int */
443 psConfig->u8DataRJumpwidth = (u32Dtq + (5 - 1)) / 5;
444 }
445 else
446 {
447 psConfig->u8DataPropSeg = 0;
448 psConfig->u8DataPhaseSeg1 = 0;
449 psConfig->u8DataPhaseSeg2 = 0;
450 psConfig->u8DataRJumpwidth = 0;
451 }
452}

467static uint32_t CANFD_CalculateTimingValues(uint32_t u32NominalBaudRate, uint32_t u32DataBaudRate, uint32_t u32SourceClock_Hz, CANFD_TIMEING_CONFIG_T *psConfig)
468{
469 int i32Nclk;
470 int i32Nclk2;
471 int i32Ntq;
472 int i32Dclk;
473 int i32Dclk2;
474 int i32Dtq;
475
476 /* observe baud rate maximums */
477 if (u32NominalBaudRate > MAX_NOMINAL_BAUDRATE) u32NominalBaudRate = MAX_NOMINAL_BAUDRATE;
478
479 for (i32Ntq = MAX_TIME_QUANTA; i32Ntq >= MIN_TIME_QUANTA; i32Ntq--)
480 {
481 i32Nclk = u32NominalBaudRate * i32Ntq;
482
483 for (psConfig->u16NominalPrescaler = 0x001; psConfig->u16NominalPrescaler <= 0x400; (psConfig->u16NominalPrescaler)++)
484 {
485 i32Nclk2 = i32Nclk * psConfig->u16NominalPrescaler;
486
487 if (((u32SourceClock_Hz / i32Nclk2) <= 5) && ((float)(u32SourceClock_Hz) / i32Nclk2) == (u32SourceClock_Hz / i32Nclk2))
488 {
489 psConfig->u8PreDivider = u32SourceClock_Hz / i32Nclk2;
490
491
492 /* if not using baudrate switch then we are done */
493 if (!u32DataBaudRate)
494 {
495 i32Dtq = 0;
496 psConfig->u8DataPrescaler = 0;
497 CANFD_GetSegments(u32NominalBaudRate, u32DataBaudRate, i32Ntq, i32Dtq, psConfig);
498 return TRUE;
499 }
500
501 /* if baudrates are the same and the solution for nominal will work for
502 data, then use the nominal settings for both */
503 if ((u32DataBaudRate == u32NominalBaudRate) && psConfig->u16NominalPrescaler <= 0x20)
504 {
505 i32Dtq = i32Ntq;
506 psConfig->u8DataPrescaler = (uint8_t)psConfig->u16NominalPrescaler;
507 CANFD_GetSegments(u32NominalBaudRate, u32DataBaudRate, i32Ntq, i32Dtq, psConfig);
508 return TRUE;
509 }
510
511 /* calculate data settings */
512 for (i32Dtq = MAX_TIME_QUANTA; i32Dtq >= MIN_TIME_QUANTA; i32Dtq--)
513 {
514 i32Dclk = u32DataBaudRate * i32Dtq;
515
516 for (psConfig->u8DataPrescaler = 0x01; psConfig->u8DataPrescaler <= 0x20; (psConfig->u8DataPrescaler)++)
517 {
518 i32Dclk2 = i32Dclk * psConfig->u8DataPrescaler;
519
520 if ((float)(u32SourceClock_Hz) / i32Dclk2 == psConfig->u8PreDivider)
521 {
522 CANFD_GetSegments(u32NominalBaudRate, u32DataBaudRate, i32Ntq, i32Dtq, psConfig);
523 return TRUE;
524 }
525 }
526 }
527
528 if (u32DataBaudRate == 0)
529 {
530 CANFD_GetSegments(u32NominalBaudRate, 0, i32Ntq, 0, psConfig);
531 return TRUE;
532 }
533 }
534 }
535 }
536
537 /* failed to find solution */
538 return 0;
539}
